		SYNTAX HIGHTER BY EXAMPLES
	
		// JavaScript
		// Different Ways (See http://goo.gl/Xz7RP2) to Define Functions:
		function a(){};             		// function declaration
		var b = function(){};       		// function expression
		var c = (function(){});     		// function expression with grouping operators
		var d = function foo(){};   		// named function expression
		var e = (function(){        		// IIFE that returns a function
					return function(){}
				})();
		var F = new Function();     		// function constructor (starts with a capital letter)
		var G = new function foo(){};   	// special case: object constructor
		var h = x => x * 2;				// ES6 arrow function
		
		// IIFE:
		(function(){console.log("Hello World!");}());   	// Pattern 1: ((){}());
		(function(){console.log("Hello World!");})();		// Pattern 2: ((){})();
		+function(){console.log("Hello World!");}();		// Pattern 3: ?(){}();
		-function(){console.log("Hello World!");}();	 	//  where ? can be		
		~function(){console.log("Hello World!");}();		//  +,-,~,!,void,etc.
		!function(){console.log("Hello World!");}();
		void function(){console.log("Hello World!");}();
		0,function(){console.log("Hello World!");}();
		(function(x){return x;}("Hello World"));
		(function(x){return x;})("Hello World");
		// IIFE with Arrow function:
		(() => {console.log("Hello World!");})();			// Pattern A: (()=>{})(); 
		(() => "Hello World!")();							// Pattern B: (()=>)();
		((x) => {return x;})("Hello World!");
		((x) => x)("Hello World!");
		(x => {return x;})("Hello World!");				// Pattern C: (=>{})();
		(x => x)("Hello World!");	  						// Pattern D: (=>)();
